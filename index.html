// Автоматическая загрузка панели или выдача чистого Base64 для Nekobox/Happ window.onload = function() { const urlParams = new URLSearchParams(window.location.search); const dataParam = urlParams.get('data'); if (dataParam) { try { const decodedData = JSON.parse(decodeURIComponent(escape(atob(dataParam)))); // Формируем чистый текст подписки с метаданными const subscriptionText = `# profile-title: ${decodedData.title} #!name=${decodedData.title} # profile-update-interval: 1 # subscription-userinfo: upload=0; download=0; total=0 ${decodedData.code}`; // ПРОВЕРКА: Если запрос идет от приложения (Nekobox, HappProxy, v2ray и т.д.) const userAgent = navigator.userAgent.toLowerCase(); const isVpnApp = userAgent.includes('nekobox') || userAgent.includes('happ') || userAgent.includes('v2ray') || userAgent.includes('clash') || userAgent.includes('sing-box'); if (isVpnApp) { // Если это приложение — кодируем ВЕСЬ текст подписки в Base64 и отдаем ТОЛЬКО его const base64Output = btoa(unescape(encodeURIComponent(subscriptionText))); document.body.innerHTML = base64Output; document.body.style.backgroundColor = "#ffffff"; document.body.style.color = "#000000"; document.body.style.fontFamily = "monospace"; document.body.style.padding = "0"; document.body.style.margin = "0"; return; // Останавливаем выполнение, чтобы HTML-панель не грузилась } // Если открыл ЧЕЛОВЕК в браузере — показываем твою красивую панель (Фото 2) document.getElementById('startScreen').style.display = 'none'; document.getElementById('generatorBlock').style.display = 'none'; document.getElementById('displayTitle').innerText = decodedData.title; document.getElementById('displayUser').innerText = decodedData.user; document.getElementById('displayDate').innerText = decodedData.date; const listContainer = document.getElementById('displayCodesList'); listContainer.innerHTML = ""; const lines = decodedData.code.split('\n'); lines.forEach(line => { if(line.trim() !== "") { const li = document.createElement('li'); li.className = "link-item"; li.innerText = line; listContainer.appendChild(li); } }); document.getElementById('finalPanel').style.display = 'block'; } catch(e) { console.error("Ошибка разбора ссылки подписки", e); } } }
